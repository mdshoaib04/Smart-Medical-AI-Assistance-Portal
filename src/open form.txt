import tkinter as tk
import webbrowser

def open_form():
    html_content = """
    <html>
    <head><title>Webinar Registration Form</title></head>
    <body>
        <div class="visme_d"
             data-title="Webinar Registration Form"
             data-url="g7ddqxx0-untitled-project?fullPage=true"
             data-domain="forms"
             data-full-page="true"
             data-min-height="100vh"
             data-form-id="133190">
        </div>
        <script src="https://static.visme.co/forms/vismeforms-embed.js"></script>
    </body>
    </html>
    """

    # Save to file
    with open("form.html", "w") as f:
        f.write(html_content)

    # Open in browser
    webbrowser.open("form.html")

# GUI setup
root = tk.Tk()
root.title("Webinar Registration")
root.geometry("300x150")

label = tk.Label(root, text="Click below to open the registration form:")
label.pack(pady=10)

button = tk.Button(root, text="Open Form", command=open_form)
button.pack(pady=10)

root.mainloop()
